package com.gestion.Hopitaux.service;

import java.util.List;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import com.gestion.Hopitaux.model.Hopital;
import com.gestion.Hopitaux.repository.HopitalRepository;
import com.gestion.Hopitaux.service.Interface.IHopitalService;

@Service
public class HopitalServiceImpl implements IHopitalService {

	
	@Autowired
	private HopitalRepository repoHopital;
	
	@Override
	public List<Hopital> findAll() {			
		return repoHopital.findAll();
	}

	@Override
	public Hopital findById(int id) {
		
		//return repoHopital.findById(id).get();
		return repoHopital.findHopitalById(id);
	}

	@Override
	public Hopital create(Hopital hopital) {		
	return repoHopital.save(hopital);
		 
	}

	@Override
	public void update(int id, Hopital hopital) {
		hopital.setId(id);		
		repoHopital.save(hopital);
		
	}
	@Override
	public void delete(int id) {
		repoHopital.deleteById(id);		
	}

	@Override
	public List<Hopital> findByVilleOrLocalite(String motcle) {		
		return repoHopital.findByVilleOrLocalite(motcle);
	}

	@Override
	public List<Hopital> findByVilleAndSpecialisationId(int specialisation, String ville) {
		return repoHopital.findHopitalByVilleAndSpecialisationId(specialisation, ville);		 
	}

	@Override
	public List<Hopital> findByVilleAndSpecialisationName(String specialisation, String ville) {		
		return repoHopital.findHopitalByVilleAndSpecialisationName(specialisation, ville);							
	}

	@Override
	public List<Hopital> findBySpecialisation(String specialisation) {		
		return repoHopital.findBySpecialisationName(specialisation);
	}
	
	
	@Override
	public List<Hopital> findBySpecialisationId(int specialisation) {		
		return repoHopital.findBySpecialisationId(specialisation);
	}
	
	

}
