package com.gestion.Hopitaux.service;

import java.util.List;
import java.util.Set;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.gestion.Hopitaux.model.Specialisation;
import com.gestion.Hopitaux.repository.HopitalRepository;
import com.gestion.Hopitaux.repository.SpecialisationRepository;
import com.gestion.Hopitaux.service.Interface.ISpecialisationService;

@Service
public class SpecialisationServiceImpl implements ISpecialisationService{

	@Autowired 
	private HopitalRepository repoHopital;
	
	@Autowired 
	private SpecialisationRepository repoSpecialisation;
	
	@Override
	public Set<Specialisation> findAllOfHopital(int id) {		
		return  repoHopital.findById(id).get().getSpecialisations();
	}

	@Override
	public Specialisation findById(int id) {
	//	return  repoSpecialisation.findById(id).get();
		return  repoSpecialisation.findSpecialisationById(id);
		 
	}
	
	@Override
	public List<Specialisation> findSpecialisationByNameOrSpecialite(String search) {
		return  repoSpecialisation.findByNameOrSpecialite(search);
		 
	}

	@Override
	public void update(int id, Specialisation specialisation) {
		specialisation.setId(id);
		repoSpecialisation.save(specialisation);
		
	}

	@Override
	public List<Specialisation> findAll() {		 
		return repoSpecialisation.findAll();
	}

	@Override
	public void delete(int id) {
		repoSpecialisation.deleteById(id);
		
	}

}
