package com.gestion.Hopitaux.helper;

import com.gestion.Hopitaux.exception.Error;
import com.gestion.Hopitaux.model.Accident;
import com.gestion.Hopitaux.model.Hopital;
import com.gestion.Hopitaux.model.Reservation;
import com.gestion.Hopitaux.model.Specialisation;

public final class CtrlControl {

	
	public static Hopital NotFoundHopital(Hopital hopital) {
		
		if(hopital == null) { throw new Error(); }
		
		return hopital;
	}
	
public static Accident NotFoundAccident(Accident accident) {
		
		if(accident == null) { throw new Error(); }
		
		return accident;
	}
	
    public static Specialisation NotFoundSpecialisation(Specialisation specialisation) {
		
		if(specialisation == null) { throw new Error(); }
		
		return specialisation;
	}
    
    
   public static Reservation NotFoundReservation(Reservation reservation) {
		
		if(reservation == null) { throw new Error(); }
		
		return reservation;
	}

      public static <T>  T NoFoundObject (T object) {
		
		if(object == null) { throw new Error(); }
		
		return object;
	}
	
}
